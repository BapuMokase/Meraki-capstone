1)  CREATE TABLE ADMIN1(AID VARCHAR(20) PRIMARY KEY NOT NULL , APASSWORD VARCHAR2(20) NOT NULL);
 
2)  CREATE TABLE USERS(USERID VARCHAR2(20) PRIMARY KEY NOT NULL,PASSWORD VARCHAR2(20),UNAME VARCHAR2(20)NOT NULL, UPHONE NUMBER(10) NOT NULL);
 
3)  CREATE TABLE CATEGORY (CATEGORYID NUMBER PRIMARY KEY NOT NULL,CATEGORYNAME VARCHAR2(30) NOT NULL,
  CATEGORYDESCRIPTION VARCHAR2(50));

 
4) CREATE TABLE PRODUCTDETAIL(PRODUCTID NUMBER PRIMARY KEY, PRODUCTNAME VARCHAR(20) NOT NULL, PRICE NUMBER NOT NULL, CATEGORYID NUMBER REFERENCES CATEGORY(CATEGORYID),COMPANY VARCHAR(20), 
  QUANTITYOH NUMBER NOT NULL, DESCRIPTION VARCHAR(20),IMAGEPATH VARCHAR(200) NOT NULL);

5)  CREATE TABLE PROMOTIONALEVENT(PROMOTIONALID NUMBER PRIMARY KEY,EVENTNAME VARCHAR2(20)NOT NULL,EVENTSTARTDATE DATE NOT NULL,
  EVENTENDDATE DATE NOT NULL,EVENTDESCRIPTION VARCHAR2(30));
 
6)  CREATE TABLE COUPONS(CID NUMBER PRIMARY KEY,CNAME VARCHAR2(30),CDESCRIPTION VARCHAR2(50),PID NUMBER REFERENCES PROMOTIONALEVENT(PROMOTIONALID),DISCOUNT NUMBER,
  CATEGORYID NUMBER REFERENCES CATEGORY (CATEGORYID));
 
7)  CREATE TABLE CART (CARTID NUMBER PRIMARY KEY ,USERID VARCHAR2(20)  REFERENCES  USERS(USERID),CARTDATE DATE NOT NULL);
 
8)  CREATE TABLE PAYMENT (PAYID NUMBER PRIMARY KEY, CARTID NUMBER  REFERENCES CART(CARTID),TOTALBILLAMOUNT NUMBER NOT NULL,
  TOTALDISCOUNT NUMBER NOT NULL,CARDNO NUMBER(30)NOT NULL,CARDCVV NUMBER NOT NULL ,CARDEXPIRY DATE NOT NULL);
  
9) CREATE TABLE CARTDETAIL(CARTDETAILID NUMBER PRIMARY KEY,CARTID NUMBER REFERENCES CART(CARTID),
  PRODUCTID NUMBER REFERENCES PRODUCTDETAIL(PRODUCTID),QUANTITY NUMBER NOT NULL);
